  5. C program to arrange 10 user input number in ascending order.

    6. 
#include<stdio.h>
#include<conio.h>
int main()
{
int n[50],t;
    for (int i = 0; i < 10; i++)
    {
        printf("\nEnter a number: ");
        scanf("%d",&n[i]);
    }

    for (int i = 0; i < 10-1; i++)
    {
        for (int j = i+1; j < 10; j++)
        {
            if (n[i]>n[j])
            {
                t=n[i];
                n[i]=n[j];
                n[j]=t;

            }
            
        }
        
    }
    printf("\nNumber in ascending order are: ");
    for (int i = 0; i < 10; i++)
    {
        printf("\n%d",n[i]);
    }
    
    
    
    return 0;
}

  11. C program to find whether the entered number is Perfect number or not.

    12. 
#include<stdio.h>
#include<conio.h>
int main()
{
    int n,sum=0;
    printf("Enter the number: ");
    scanf("%d",&n);
    for (int i = 1; i < n; i++)
    {
        if (n%i==0)
        {
            sum=sum+i;
        }
    }
    if (sum==n)
    {
        printf("\nPerfect number");
    }
    else
    {
        printf("\nNot perfect");
    }
    
    
    return 0;
}
